# Example for adding time derived measurements

See `add_time_derived_measurements.py` for the source of the constant value.

```python
from datetime import datetime

seconds_in_year = 31556926.080000002
times_by_dob = {
    "12/28/1980, 00:00:00": [
        "05/11/2010, 17:41:51",
        "05/11/2010, 17:48:48",
        "05/11/2010, 18:25:35",
        "05/11/2010, 18:57:18",
        "05/11/2010, 19:27:19",
    ],
    "06/20/1978, 00:00:00": [
        "06/20/2010, 19:23:52",
        "06/20/2010, 19:25:32",
        "06/20/2010, 19:45:19",
        "06/20/2010, 20:12:31",
        "06/20/2010, 20:24:44",
        "06/20/2010, 20:41:33",
        "06/20/2010, 20:50:04",
    ],
    "03/09/1978, 00:00:00": [
        "05/26/2010, 02:30:56",
        "05/26/2010, 04:51:52",
    ],
    "03/28/1976, 00:00:00": [
        "02/05/2010, 05:55:39",
        "02/05/2010, 07:02:30",
    ],
    "12/19/1988, 00:00:00": [
        "01/03/2010, 06:27:59",
        "01/03/2010, 08:22:13",
    ],
    "07/20/1986, 00:00:00": [
        "06/03/2010, 14:54:38",
        "06/03/2010, 15:39:49",
        "06/03/2010, 16:20:49",
        "06/03/2010, 16:44:26",
    ],
}

ts_fmt = "%m/%d/%Y, %H:%M:%S"

ages_by_dob = {}
for k, ts in times_by_dob.items():
    print(k)
    dob_ts = datetime.strptime(k, ts_fmt)
    for event_ts_str in ts:
        event_ts = datetime.strptime(event_ts_str, ts_fmt)
        age = (event_ts - dob_ts).total_seconds() / seconds_in_year
        print(f"  - {age}")
```
